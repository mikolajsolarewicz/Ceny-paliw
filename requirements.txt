"""
generate_sample.py
-------------------
Tworzy REALISTYCZNY, ale SYNTETYCZNY zestaw danych w data/, żeby dashboard
działał od razu po sklonowaniu repo i był odporny na chwilową niedostępność API
podczas prezentacji.

Dane odwzorowują rzeczywiste zjawiska z lat 2020–2026 (krach COVID, skok cen
2022, tarcza VAT, stabilizacja 2024–2026) i celowo zawierają OPÓŹNIENIE oraz
ASYMETRIĘ reakcji cen na stacjach, tak by analiza w dashboardzie miała co pokazać.

>>> To NIE są prawdziwe dane. Przed oddaniem projektu uruchom: python fetch_data.py
"""

from __future__ import annotations

import json
from pathlib import Path

import numpy as np
import pandas as pd

DATA_DIR = Path(__file__).parent / "data"
DATA_DIR.mkdir(exist_ok=True)
LITERS_PER_BARREL = 158.987
rng = np.random.default_rng(42)


def _interp(anchors: dict[str, float], dates: pd.DatetimeIndex) -> np.ndarray:
    """Liniowa interpolacja między punktami kotwiczącymi (data->wartość)."""
    ax = pd.to_datetime(list(anchors.keys())).astype("int64").to_numpy()
    ay = np.array(list(anchors.values()), dtype=float)
    dx = dates.astype("int64").to_numpy()
    return np.interp(dx, ax, ay)


def _ar1_noise(n: int, sd: float, rho: float = 0.92) -> np.ndarray:
    """Szum AR(1) – nadaje seriom realistyczną 'lepkość' zamiast białego szumu."""
    e = rng.normal(0, sd, n)
    out = np.zeros(n)
    for i in range(1, n):
        out[i] = rho * out[i - 1] + e[i]
    return out


def build() -> None:
    dates = pd.date_range("2020-01-01", "2026-06-06", freq="D")
    n = len(dates)

    # --- Brent (USD/bbl): krach COVID 2020, skok 2022, stabilizacja --------- #
    brent = _interp({
        "2020-01-01": 64, "2020-04-21": 20, "2020-12-01": 50,
        "2021-06-01": 73, "2021-12-01": 75, "2022-03-08": 124,
        "2022-06-01": 117, "2022-09-01": 90, "2022-12-01": 81,
        "2023-06-01": 75, "2023-12-01": 77, "2024-06-01": 82,
        "2024-12-01": 73, "2025-06-01": 70, "2025-12-01": 72,
        "2026-06-06": 71,
    }, dates) + _ar1_noise(n, 0.6)
    brent = np.clip(brent, 15, None)

    # --- Kurs USD/PLN ------------------------------------------------------- #
    usd = _interp({
        "2020-01-01": 3.80, "2020-04-01": 4.18, "2020-12-01": 3.75,
        "2021-12-01": 4.05, "2022-03-01": 4.30, "2022-10-15": 5.02,
        "2022-12-01": 4.40, "2023-06-01": 4.10, "2023-12-01": 3.97,
        "2024-06-01": 4.00, "2024-12-01": 4.10, "2025-06-01": 3.80,
        "2026-06-06": 3.78,
    }, dates) + _ar1_noise(n, 0.012)

    crude_pln_l = brent * usd / LITERS_PER_BARREL  # surowiec w PLN/l

    # --- "Tarcza" VAT 8% na paliwa (real: ~luty–grudzień 2022) -------------- #
    tarcza = np.where(
        (dates >= "2022-02-01") & (dates <= "2022-12-31"), -0.70, 0.0
    )

    # --- Ceny detaliczne: model częściowego dostosowania z LAGiem ----------- #
    # cel(t) = baza + przełożenie * surowiec(t-lag) + akcyza/marża + sezon
    # cena(t) = cena(t-1) + tempo * (cel(t) - cena(t-1)),  tempo asymetryczne
    LAG = 11
    crude_lagged = np.concatenate([np.full(LAG, crude_pln_l[0]), crude_pln_l[:-LAG]])
    doy = dates.dayofyear.to_numpy()
    season_summer = 0.10 * np.sin(2 * np.pi * (doy - 80) / 365.25)   # latem drożej (95)
    season_winter = 0.12 * np.cos(2 * np.pi * (doy) / 365.25)        # zimą drożej (ON)

    def simulate(base: float, passthrough: float, season: np.ndarray,
                 up_speed: float, down_speed: float, noise_sd: float) -> np.ndarray:
        target = base + passthrough * crude_lagged + season + tarcza
        price = np.zeros(n)
        price[0] = target[0]
        for i in range(1, n):
            gap = target[i] - price[i - 1]
            speed = up_speed if gap > 0 else down_speed   # rakieta (szybko w górę) / piórko (wolno w dół)
            price[i] = price[i - 1] + speed * gap
        return price + _ar1_noise(n, noise_sd, rho=0.6)

    pb95 = simulate(3.70, 1.45, season_summer, up_speed=0.16, down_speed=0.07, noise_sd=0.015)
    on = simulate(4.30, 1.58, season_winter, up_speed=0.17, down_speed=0.08, noise_sd=0.018)
    pb98 = pb95 + 0.82 + _ar1_noise(n, 0.01, rho=0.6)

    fuel = pd.DataFrame({
        "date": dates,
        "PB95": pb95.round(3),
        "PB98": pb98.round(3),
        "ON": on.round(3),
    })

    # Kurs i Brent zapisujemy tylko dla dni roboczych (jak prawdziwe źródła)
    business = dates[dates.weekday < 5]
    usd_df = pd.DataFrame({"date": business,
                           "usdpln": pd.Series(usd, index=dates).loc[business].round(4).to_numpy()})
    brent_df = pd.DataFrame({"date": business,
                             "brent_usd": pd.Series(brent, index=dates).loc[business].round(2).to_numpy()})

    fuel.to_csv(DATA_DIR / "fuel_prices.csv", index=False)
    usd_df.to_csv(DATA_DIR / "usdpln.csv", index=False)
    brent_df.to_csv(DATA_DIR / "brent.csv", index=False)
    (DATA_DIR / "_meta.json").write_text(json.dumps({
        "source": "sample",
        "note": "Dane syntetyczne. Uruchom fetch_data.py po prawdziwe dane.",
        "generated_at": pd.Timestamp.now().isoformat(timespec="seconds"),
    }, indent=2))
    print(f"✓ Wygenerowano przykładowe dane: {n} dni, "
          f"{fuel['date'].min().date()} – {fuel['date'].max().date()}")
    print(f"   PB95 dziś≈{pb95[-1]:.2f}  ON≈{on[-1]:.2f}  PB98≈{pb98[-1]:.2f} zł/l")


if __name__ == "__main__":
    build()
